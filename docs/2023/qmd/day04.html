Count the number of winning cards \(w\) and return a score \(s\), defined as \[s=\begin{cases} 0, \text{ if } w = 0 \\ 2^{w-1}, \text{ if } w > 0 \end{cases}\]

  • R (HW)
  • Python
matches = function(x){
  
  spl = strsplit(x, split = "\\|")[[1]]
  pairs = lapply(
    1:2
    , FUN = \(i){
      strsplit(spl[i], split = " ")[[1]] |> 
        as.integer() |> 
        na.omit()
    }
  )
  
  sum(pairs[[2]] %in% pairs[[1]])
  
}

txt = gsub("^.*?:", "", txt) 

m = Map(
  matches
  , x = txt
) |> 
  unname() |> 
  unlist()
